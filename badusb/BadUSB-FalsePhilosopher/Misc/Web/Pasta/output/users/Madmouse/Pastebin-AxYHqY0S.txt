void encode_instruction(std::vector<int>* bcode, std::vector<std::string> c)
{
	int in = instructions.at(c.front())[1];
	for(int i=(Madnsize(in)/SIZE_BYTE)-1;i>=0;i--)
		bcode->push_back((in&(0xff<<SIZE_BYTE*i))>>SIZE_BYTE*i);
}
