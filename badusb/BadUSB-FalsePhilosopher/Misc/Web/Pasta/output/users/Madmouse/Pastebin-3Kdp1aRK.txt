echo -ne '\n\n// C/C++\nconst char evil[] = "';
for i in $(objdump -d $1 -M intel |grep "^ " |cut -f2);
	do echo -n '\x'$i;
done;echo -e '";\n\n'

echo -ne '# Python\nevil = "';
for i in $(objdump -d $1 -M intel |grep "^ " |cut -f2);
	do echo -n '\x'$i;
done;echo -e '"\n\n'

echo -ne '# Bash\n./shellcode $(echo -en "';
for i in $(objdump -d $1 -M intel |grep "^ " |cut -f2);
	do echo -n '\x'$i;
done;echo -e '")\n\n'
